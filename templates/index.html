配置预览 (stream-sni.conf)

stream {
    map $ssl_preread_server_name $backend_name {
        {{ r.domain }}    {{ r.backend_name }};
        default          default_backend;
    }

    upstream {{ r.backend_name }} {
        server {{ r.backend_server }};
    }

    upstream default_backend {
        server {{ settings.default_backend_server }};
    }

    server {
        listen {{ settings.listen_port }} reuseport;
        proxy_pass $backend_name;
        ssl_preread on;
    }
}
